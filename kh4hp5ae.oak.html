备份

diskutil unmountDisk /dev/disk2
sudo dd if=/dev/rdisk2  bs=1m  | bzip2 > backup.img.bz2
diskutil eject /dev/disk2

还原

diskutil unmountDisk /dev/disk2
bzip2 -dc backup.img.bz2 | dd of=/dev/rdisk2 bs=1m
diskutil eject /dev/disk2
