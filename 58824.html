服务器时间不准会导致很多奇怪问题，比如七牛上传文件校验token失败。因此我写了一个批处理脚本一直在后台运行，每天校准一次时间。

/tmp/date.sh

for((;;))
do
  str=`curl -I -s baidu.com|grep Date 2>/dev/null`
  echo `date -s "${str:5:-1}"`
  sleep 86400
done

启动命令 nohup /bin/bash /tmp/date.sh > /dev/null&
