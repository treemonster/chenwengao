/usr/bin/ld.bfd.real: ext/intl/msgformat/msgformat_helpers.o: undefined reference ...

/usr/bin/ld.bfd.real: note: '__gxx_personality_v0@@CXXABI_1.3' is defined in DSO ...

sed -i '/^BUILD_/ s/\$(CC)/\$(CXX)/g' Makefile
