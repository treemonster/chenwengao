n个猴子选大王，规则是围成一圈，每数到m个猴子踢出去，剩下的最后一个是大王。

let n=100,m=15

let o=[]
for(i=0;i<n;i++){
  o.push(i)
}
for(i=0;o.length>1;){
  i=(i+m)%o.length
  o.splice(i,1)
}
alert(o[0])
