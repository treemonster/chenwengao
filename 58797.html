@echo off
echo 文本扫描器
echo 作者：雾海树妖^<admin@xdelve.com^>
if "%2" neq "" for /f "tokens=*" %%i in ('dir/s/b/a:-D %1') do (
 type "%%i"|findstr /I %2 >nul 2>nul&&echo %%i
 echo "%%i"|findstr /I %2 >nul 2>nul&&echo %%i
) else echo 使用方法： "%0" "目标文件夹路径" "扫描关键字" && pause
