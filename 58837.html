记录开发生涯的踩坑经历，用时间来验证成长

windows总是把一些用得到的功能藏得很深。。比如看开机时间，为了不用再特地进事件管理器找开机时间，我写了一个mshta脚本用于自动获取。

保存为a.hta。

<!--
author: treemonster<https://treemonster.github.io/>
date: 2017-02-22
-->
<script language="Javascript">
lastdate=(new Date).getDate();
lasttime=null;
function slice(a,b,c){ return Array.prototype.slice.call(a,b,c);}
window.checktime=function(a){
  var m;
  a.replace(/^(....)(..)(..)(..)(..)(..)/g,function(){
    var a=arguments;
    if(a[3]==lastdate) return lasttime=a;
    m=true;
    a=lasttime;
    a[4]=a[4]*1+8; // China time zone +8 hours
    alert('today open time: '+slice(a,4,7).join(':'));
    window.close();
  });
  return m;
};
</script>

<script language="VBScript">
Set objWMIService=GetObject("winmgmts://./root/cimv2")
Set li = objWMIService.ExecQuery("Select * from Win32_NTLogEvent where Logfile='Security'",,16)
For Each EventLogEntry in li
  if window.checktime(EventLogEntry.TimeGenerated) then
    exit for
  end if
next
</script>
