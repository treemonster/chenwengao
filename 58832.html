module.exports=(generator,context)=>function(){
  const gen=generator.apply(context||{},arguments);
  var def=Promise.defer();
  void function next(){
    try{
      // fixed by treemonster@ 2017-08-21 <https://xdelve.com>
      // on msie, raw arguments array cannot use. this is also a bug on babel.js, but by this way is ok
      let c=gen.next(Array.prototype.slice.call(arguments))
      if(c.done) def.resolve(c.value);
      else if(typeof c.value==='function')c.value(next);
      else if(c.value && c.value.constructor===Promise) c.value.then(next);
      else setTimeout(next,10);
    }catch(e){def.reject(e)}
  }();
  return def.promise;
};
